```bash
#!/bin/bash
set -e

echo "== VoulezVous - Setup Automático =="

# 1. Checar dependências mínimas
command -v docker >/dev/null 2>&1 || { echo >&2 "Docker não encontrado. Instale antes de prosseguir."; exit 1; }
command -v docker-compose >/dev/null 2>&1 || { echo >&2 "Docker Compose não encontrado. Instale antes."; exit 1; }

# 2. Subir bancos e serviços de apoio
echo "[1/5] Subindo serviços com Docker Compose..."
docker-compose up -d

# 3. Rodar migrations e seeds
echo "[2/5] Aplicando schema e seed no Supabase/Postgres..."
docker cp supabase/schema.sql postgres:/schema.sql
docker cp supabase/seed.sql postgres:/seed.sql
docker exec -i postgres psql -U postgres -f /schema.sql
docker exec -i postgres psql -U postgres -f /seed.sql

# 4. Instalar dependências backend
echo "[3/5] Instalando dependências Go (backend)..."
cd backend && go mod tidy && cd ..

# 5. Instalar dependências frontend
echo "[4/5] Instalando dependências React (frontend)..."
cd frontend && yarn install && cd ..

# 6. Rodar tudo
echo "[5/5] Iniciando backend e frontend..."
cd backend && go run main.go &
cd frontend && yarn start &

echo "== Tudo pronto! Acesse http://localhost:3000 =="
```
