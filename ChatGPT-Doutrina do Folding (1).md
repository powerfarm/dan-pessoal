//! Exemplo prático: Dobrando uma trajetória linear simples (.traj) em FTO ótimo
//! Demonstra as quatro dobras em ação.

use logline_runtime::folding_engine::{
    LinearSequence, DependencyGraph, ResourceTopography, 
    StateFieldSimulator, FoldingOptimizer, FoldedTrajectoryObject
};

fn main() -> Result<(), Box<dyn std::error::Error>> {
    println!("🧬 Iniciando Demonstração do Folding Engine");
    println!("⸻");
    
    // 🎯 INPUT: Sequência Linear Bruta (PTO - Plano de Trajetória Original)
    let traj_content = r#"
    name: "Projeto Alpha"
    steps:
      - id: A
        desc: "Aquisição de dados"
        duration: 2h
        deps: []
        resources: { cpu: 2, memory: 4, human: 1 }
        
      - id: B  
        desc: "Pré-processamento"
        duration: 1h
        deps: [A]
        resources: { cpu: 4, memory: 8, human: 1 }
        
      - id: C
        desc: "Análise estatística"
        duration: 3h
        deps: [B]
        resources: { cpu: 8, memory: 16, human: 2 }
        
      - id: D
        desc: "Geração de relatório"
        duration: 1h
        deps: [C]
        resources: { cpu: 2, memory: 2, human: 1 }
        
      - id: E
        desc: "Validação externa"
        duration: 4h
        deps: [A]  // ⚡ PODE COMEÇAR APÓS A, EM PARALELO COM B e C!
        resources: { network: 1, human: 1 }
    "#;
    
    println!("📜 DOBRA 1: Sequência Linear (.traj)");
    println!("{}", traj_content);
    
    // Parsing da sequência linear
    let linear_seq = LinearSequence::parse_yaml(traj_content)?;
    println!("✅ Sequência parseada: {} steps", linear_seq.steps().len());
    
    // 🎯 DOBRA 1: Grafo de Dependência (DAG)
    let dag = DependencyGraph::build(&linear_seq)?;
    println!("\n🔷 DOBRA 2: Grafo de Dependência");
    println!("Nós: {}", dag.nodes().len());
    println!("Arestas: {}", dag.edges().len());
    println!("Caminho crítico: {:?}", dag.critical_path());
    
    // 🎯 DOBRA 2: Topografia de Recursos
    let resource_map = ResourceTopography::new()
        .with_cpu_pool(16)
        .with_memory_pool(32)
        .with_human_pool(3)
        .with_network_bandwidth(1.0);
        
    println!("\n🗺️  DOBRA 3: Topografia de Recursos");
    println!("CPU disponível: {}", resource_map.cpu_pool());
    println!("Memória: {}", resource_map.memory_pool());
    println!("Humanos: {}", resource_map.human_pool());
    
    // 🎯 DOBRA 3: Espaço de Estados Temporal
    let mut state_simulator = StateFieldSimulator::new(&dag, resource_map);
    state_simulator.simulate_causal_effects()?;
    
    println!("\n🌀 DOBRA 4: Campo de Estados Temporais");
    println!("Estados simulados: {}", state_simulator.state_count());
    println!("Custos de transição mapeados");
    
    // 🎯 DOBRA 4: Cálculo da Geodésica Ótima
    let geodesic_solver = GeodesicSolver::new(state_simulator);
    let optimal_path = geodesic_solver.find_optimal_geodesic()?;
    
    println!("\n⚡ TRAJETÓRIA ÓTIMA ENCONTRADA");
    println!("Ordem original: A → B → C → D | E paralelo?");
    println!("Ordem ótima: {}", optimal_path);
    
    // 🎯 FOLDING FINAL: Geração do FTO
    let folding_optimizer = FoldingOptimizer::new();
    let fto = folding_optimizer.fold_trajectory(linear_seq, optimal_path)?;
    
    println!("\n💎 FTO GERADO (Folded Trajectory Object)");
    println!("Hash: {}", fto.hash());
    println!("DeltaS (economia): {:.2}%", fto.delta_s() * 100.0);
    println!("Juice Factor: {:.3}", fto.juice_factor());
    println!("Curva de recursos: {:?}", fto.resource_curve());
    
    // Demonstração da execução adaptativa
    println!("\n🚀 EXECUÇÃO DOBRADA");
    for (i, step) in fto.adaptive_order().iter().enumerate() {
        println!("{}. {} [Recursos: {:?}]", i + 1, step.id, step.allocated_resources);
    }
    
    Ok(())
}

//! Representação da sequência linear bruta (.traj file)
//! Corresponde ao PTO (Plano de Trajetória Original)

use serde::Deserialize;
use std::collections::HashMap;

#[derive(Debug, Deserialize, Clone)]
pub struct LinearSequence {
    pub name: String,
    pub steps: Vec<SequenceStep>,
}

#[derive(Debug, Deserialize, Clone)]
pub struct SequenceStep {
    pub id: String,
    pub desc: String,
    pub duration: String, // Ex: "2h", "30min"
    pub deps: Vec<String>,
    pub resources: HashMap<String, f64>,
}

impl LinearSequence {
    /// Parse a YAML string into a LinearSequence
    pub fn parse_yaml(yaml_content: &str) -> Result<Self, serde_yaml::Error> {
        serde_yaml::from_str(yaml_content)
    }
    
    /// Get all steps in the sequence
    pub fn steps(&self) -> &[SequenceStep] {
        &self.steps
    }
    
    /// Find a step by ID
    pub fn find_step(&self, id: &str) -> Option<&SequenceStep> {
        self.steps.iter().find(|step| step.id == id)
    }
}

# 🧬 Folding Engine

O motor de folding é o núcleo computacional do LogLineOS, responsável por transformar trajetórias lineares (PTO) em estruturas causais ótimas (FTO) através de quatro dobras sequenciais.

## As Quatro Dobras

1. **Sequência Linear** (`sequence.rs`) - Parse do arquivo `.traj`
2. **Grafo de Dependência** (`dag_builder.rs`) - Análise de paralelismo causal  
3. **Topografia de Recursos** (`resource_topography.rs`) - Mapeamento de restrições
4. **Espaço de Estados Temporal** (`state_field.rs`) - Simulação de efeitos causais

## Saída: FTO (Folded Trajectory Object)

O FTO é a trajetória matematicamente ótima, contendo:
- `hash`: Identidade única causal
- `deltaS`: Economia de esforço total
- `juice_factor`: Benefício causal cruzado
- `adaptive_order`: Ordem de execução não-linear
- `resource_curve`: Curva de alocação ótima

## Exemplo de Usorust
use folding_engine::{LinearSequence, FoldingOptimizer};

// Carregar trajetória linear
let linear = LinearSequence::parse_yaml(traj_content)?;

// Dobrar em FTO ótimo
let fto = FoldingOptimizer::new().fold_trajectory(linear)?;

// Executar trajetória dobrada
for step in fto.adaptive_order() {
    execute_step(step)?;
}
