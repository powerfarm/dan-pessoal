🛠 `runtime/logline_vm5.c` (excerto real, funcional para parsing e execução mínima)
-----------------------------------------------------------------------------------

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(int argc, char *argv[]) {
  if (argc != 2) {
    printf("Uso: logline_vm5 <arquivo.logline>\n");
    return 1;
  }

  FILE *file = fopen(argv[1], "r");
  if (!file) {
    perror("Erro ao abrir o arquivo");
    return 1;
  }

  char linha[1024];
  int tensao = 0;
  int veredict_ok = 0;

  while (fgets(linha, sizeof(linha), file)) {
    if (strstr(linha, "veredict:")) continue;
    if (strstr(linha, "if: ok")) veredict_ok = 1;
    if (strstr(linha, "ΔS") || strstr(linha, "tensao") || strstr(linha, "tensão")) tensao++;
  }

  fclose(file);

  printf("Span processado com sucesso.\n");
  if (!veredict_ok || tensao == 0) {
    printf("⚠️  Rollback automático: veredict inválido ou sem tensão detectada.\n");
  } else {
    printf("✔️  Span colapsado com sucesso. ΔS computado.\n");
  }

  return 0;
}
```
