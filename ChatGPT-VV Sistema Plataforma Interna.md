### 3.2. Fluxo de Login/Logout

Um componente simples de login usando Supabase:

```jsx
// src/components/Login.js
import React, { useState } from 'react'
import { supabase } from '../supabaseClient'

function Login() {
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [error, setError] = useState(null)

  const handleLogin = async (e) => {
    e.preventDefault()
    const { user, error } = await supabase.auth.signInWithPassword({
      email,
      password,
    })
    if (error) setError(error.message)
  }

  return (
    <div>
      <h1>Login VV Sistema</h1>
      <form onSubmit={handleLogin}>
        <input
          type="email"
          placeholder="Seu e-mail"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
        />
        <input
          type="password"
          placeholder="Sua senha"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
        />
        <button type="submit">Entrar</button>
      </form>
      {error && <p>{error}</p>}
    </div>
  )
}

export default Login
```

Você pode, claro, implementar MFA ou “RFID login” de forma mais avançada, mas isso geralmente envolve backend/hardware adicional (ex. leitores RFID que comuniquem com a API para validar o token).

4\. Criando o “Layout Drag-and-Drop” (Exemplo Simplificado)
-----------------------------------------------------------

### 4.1. Estrutura de Dados

Você pode ter uma tabela no Supabase chamada `user_layouts` que armazena a configuração de cada usuário, por exemplo:

```sql
create table if not exists user_layouts (
  id bigserial primary key,
  user_id uuid references auth.users not null,
  layout jsonb not null,   -- ou text para armazenar em JSON
  created_at timestamp default now()
);
```

No campo `layout`, vamos guardar algo como:

```json
[
  { "component": "OrdersTable", "x": 0, "y": 0 },
  { "component": "StockCard", "x": 0, "y": 1 },
  ...
]
```

Ou seja, cada bloco (component) que o usuário adicionou no dashboard, junto com a posição (x, y) ou qualquer outro metadado que você queira.

### 4.2. Criando os Blocos

Digamos que você tenha alguns componentes prontos:

*   `OrdersTable.js` (exibe pedidos)
*   `StockCard.js` (exibe status de estoque)
*   `ScheduleCard.js` (exibe turnos/horários)

Você pode criar um “registry” (ou dicionário) que mapeie o nome do componente para o componente em si:

```jsx
// src/components/blocks/index.js
import OrdersTable from './OrdersTable'
import StockCard from './StockCard'
import ScheduleCard from './ScheduleCard'

export const blockRegistry = {
  OrdersTable,
  StockCard,
  ScheduleCard
}
```

### 4.3. Container com Drag-and-Drop

Vamos supor que usamos `react-beautiful-dnd`. Você teria algo assim:

```jsx
// src/components/Dashboard.js
import React, { useEffect, useState } from 'react'
import { DragDropContext, Droppable, Draggable } from 'react-beautiful-dnd'
import { supabase } from '../supabaseClient'
import { blockRegistry } from './blocks'

function Dashboard({ userId }) {
  const [layout, setLayout] = useState([])

  useEffect(() => {
    // Carrega layout do Supabase
    const fetchLayout = async () => {
      const { data, error } = await supabase
        .from('user_layouts')
        .select('layout')
        .eq('user_id', userId)
        .single()
      if (!error && data) {
        setLayout(data.layout)
      }
    }
    fetchLayout()
  }, [userId])

  const onDragEnd = (result) => {
    // Lógica para reorder (caso o usuário arraste algum bloco na lista)
    if (!result.destination) return

    const items = Array.from(layout)
    const [reorderedItem] = items.splice(result.source.index, 1)
    items.splice(result.destination.index, 0, reorderedItem)

    setLayout(items)
    // Salvar nova ordem no Supabase
    supabase.from('user_layouts').upsert({
      user_id: userId,
      layout: items
    }).then(({ error }) => {
      if(error) console.error(error)
    })
  }

  return (
    <DragDropContext onDragEnd={onDragEnd}>
      <Droppable droppableId="dashboard">
        {(provided) => (
          <div {...provided.droppableProps} ref={provided.innerRef}>
            {layout.map((block, index) => {
              const Component = blockRegistry[block.component]
              return (
                <Draggable key={index} draggableId={`block-${index}`} index={index}>
                  {(provided) => (
                    <div
                      ref={provided.innerRef}
                      {...provided.draggableProps}
                      {...provided.dragHandleProps}
                      style={{
                        border: '1px solid #ccc',
                        padding: '8px',
                        marginBottom: '8px',
```
