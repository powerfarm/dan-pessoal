import requests

SUPABASE_URL = "https://xxxx.supabase.co"
SUPABASE_KEY = "YOUR_SECRET_KEY"

def processar_pedido(pedido_id):
    """ Processa automaticamente um pedido, atualizando status e enviando confirmação. """
    url = f"{SUPABASE_URL}/rest/v1/pedidos?id=eq.{pedido_id}"
    headers = {"apikey": SUPABASE_KEY, "Content-Type": "application/json"}
    
    pedido = requests.get(url, headers=headers).json()
    
    if pedido:
        data = {"status": "processando"}
        requests.patch(url, json=data, headers=headers)
        return "Pedido atualizado."
    return "Pedido não encontrado."
```

from flask import Flask, request, jsonify
import requests

app = Flask(__name__)

SUPABASE_URL = "https://xxxx.supabase.co"
SUPABASE_KEY = "YOUR_SECRET_KEY"

def obter_status_pedido(pedido_id):
    """ Consulta o status de um pedido no banco de dados. """
    url = f"{SUPABASE_URL}/rest/v1/pedidos?id=eq.{pedido_id}"
    headers = {"apikey": SUPABASE_KEY, "Content-Type": "application/json"}
    
    response = requests.get(url, headers=headers).json()
    return response[0]['status'] if response else "Pedido não encontrado."

@app.route("/whatsapp", methods=["POST"])
def whatsapp_webhook():
    """ Processa mensagens recebidas no WhatsApp. """
    dados = request.json
    mensagem = dados.get("mensagem", "").lower()

    if "status do pedido" in mensagem:
        pedido_id = mensagem.split("#")[-1]
        status = obter_status_pedido(pedido_id)
        resposta = f"Seu pedido está {status}."
    elif "comprar" in mensagem:
        resposta = "Deseja adicionar ao carrinho? Responda com SIM ou NÃO."
    else:
        resposta = "Desculpe, não entendi. Como posso ajudar?"

    return jsonify({"resposta": resposta})

if __name__ == "__main__":
    app.run(port=5000)
```
