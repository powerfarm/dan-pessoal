🟩 `server.js`
--------------

```js
const express = require('express');
const fs = require('fs');
const path = require('path');
const crypto = require('crypto');

const app = express();
app.use(express.json());

const LOG_PATH = path.join(__dirname, 'log.jsonl');
const SPANS_DIR = path.join(__dirname, 'spans');
if (!fs.existsSync(SPANS_DIR)) fs.mkdirSync(SPANS_DIR);

// UTIL: gera ID único
function generateId() {
  return crypto.randomBytes(6).toString('hex');
}

// POST /minicontrato
app.post('/minicontrato', (req, res) => {
  const { who, did, when, with: comQuem, witness } = req.body;
  const id = generateId();
  const timestamp = new Date().toISOString();

  const span = {
    type: "minicontrato",
    id,
    who,
    did,
    when,
    with: comQuem,
    witness,
    timestamp
  };

  const spanPath = path.join(SPANS_DIR, `${id}.json`);
  fs.writeFileSync(spanPath, JSON.stringify(span, null, 2));
  fs.appendFileSync(LOG_PATH, JSON.stringify(span) + '\n');

  res.json({ status: "ok", id });
});

// GET /ping
app.get('/ping', (req, res) => res.send("LogLine vivo"));

app.listen(5044, () => {
  console.log("🟢 LogLine executor rodando em http://localhost:5044");
});
```
