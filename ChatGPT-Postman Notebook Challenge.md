package builtins

type BuiltinFn func(args ...interface{}) (interface{}, error)

// NewBuiltinsMap retorna um mapa de todas as funções built-in disponíveis.
func NewBuiltinsMap() map[string]BuiltinFn {
	return map[string]BuiltinFn{
		"now":      Now,
		"len":      Len,
		"upper":    Upper,
		"add_days": AddDays,
		"sha256":   SHA256,
		"format":   Format,
	}
}

// Now retorna o timestamp UTC atual no formato ISO 8601.
func Now(args ...interface{}) (interface{}, error) {
	if len(args) != 0 {
		return nil, fmt.Errorf("função 'now' não aceita argumentos")
	}
	return time.Now().UTC().Format(time.RFC3339), nil
}

// Len retorna o comprimento de uma string ou slice.
func Len(args ...interface{}) (interface{}, error) {
	if len(args) != 1 {
		return nil, fmt.Errorf("função 'len' requer 1 argumento")
	}
	switch v := args[0].(type) {
	case string:
		return len(v), nil
	case []interface{}:
		return len(v), nil
	case []string:
		return len(v), nil
	default:
		return nil, fmt.Errorf("tipo não suportado para 'len': %T", args[0])
	}
}

// Upper converte uma string para maiúsculas.
func Upper(args ...interface{}) (interface{}, error) {
	if len(args) != 1 || reflect.TypeOf(args[0]).Kind() != reflect.String {
		return nil, fmt.Errorf("função 'upper' requer 1 argumento string")
	}
	return strings.ToUpper(args[0].(string)), nil
}

// AddDays adiciona um número de dias a uma data ISO 8601.
func AddDays(args ...interface{}) (interface{}, error) {
	if len(args) != 2 {
		return nil, fmt.Errorf("função 'add_days' requer 2 argumentos: data e dias")
	}
	dateStr, ok1 := args[0].(string)
	daysF,  ok2 := args[1].(float64)
	if !ok1 || !ok2 {
		return nil, fmt.Errorf("argumentos inválidos para 'add_days'")
	}
	t, err := time.Parse(time.RFC3339, dateStr)
	if err != nil {
		return nil, fmt.Errorf("data inválida '%s': %w", dateStr, err)
	}
	newDate := t.Add(time.Duration(int(math.Round(daysF))) * 24 * time.Hour)
	return newDate.Format(time.RFC3339), nil
}

// SHA256 calcula o hash SHA256 de uma string.
func SHA256(args ...interface{}) (interface{}, error) {
	if len(args) != 1 || reflect.TypeOf(args[0]).Kind() != reflect.String {
		return nil, fmt.Errorf("função 'sha256' requer 1 argumento string")
	}
	sum := sha256.Sum256([]byte(args[0].(string)))
	return fmt.Sprintf("%x", sum), nil
}

// Format substitui o primeiro "{}" pelo valor fornecido.
func Format(args ...interface{}) (interface{}, error) {
	if len(args) < 2 {
		return nil, fmt.Errorf("função 'format' requer pelo menos 2 argumentos")
	}
	s, ok := args[0].(string)
	if !ok {
		return nil, fmt.Errorf("primeiro argumento de 'format' deve ser string")
	}
	return strings.Replace(s, "{}", fmt.Sprintf("%v", args[1]), 1), nil
}

// logline-runtime/auditor/auditor.go
package auditor

import (
	"crypto/hmac"
	"crypto/sha256"
	"encoding/json"
	"fmt"
	"log"
	"time"

	"logline-runtime/registry"
)

// LogLine é o mesmo tipo de registry.LogLine para consistência.
type LogLine registry.LogLine

// Client grava LogLines no Supabase, adicionando HMAC se houver chave.
type Client struct {
	registryClient *registry.Client
	auditKey       []byte
}

// NewClient cria o cliente auditor, convertendo a chave_string para []byte.
func NewClient(regClient *registry.Client, auditKeyStr string) *Client {
	return &Client{registryClient: regClient, auditKey: []byte(auditKeyStr)}
}

// StoreAuditLog serializa, assina e envia ao registry.Client.
func (c *Client) StoreAuditLog(logLine LogLine) error {
	// Faça uma cópia sem a assinatura existente
	copyLL := logLine
	if copyLL.Details == nil {
		copyLL.Details = make(map[string]interface{})
	}
	delete(copyLL.Details, "signature")

	// Serializa para JSON (para HMAC)
	payload, err := json.Marshal(copyLL)
	if err != nil {
		return fmt.Errorf("erro ao serializar LogLine: %w", err)
	}

	// Gera HMAC se tiver chave
	if len(c.auditKey) > 0 {
		mac := hmac.New(sha256.New, c.auditKey)
		mac.Write(payload)
		sig := fmt.Sprintf("hmac-sha256:%x", mac.Sum(nil))
		logLine.Details["signature"] = sig
	}

	// Converte e envia
	return c.registryClient.StoreAuditLog(registry.LogLine(logLine))
}
