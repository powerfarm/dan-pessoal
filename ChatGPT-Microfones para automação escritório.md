# micwatch.py
import sounddevice as sd
import numpy as np
import subprocess
import queue
import time

THRESHOLD = 0.02       # nível mínimo de som para acionar
SILENCE_SECONDS = 1.5  # tempo de silêncio para reset
CHUNK_DURATION = 0.5   # segundos por amostra
SAMPLE_RATE = 16000

q_audio = queue.Queue()

def audio_callback(indata, frames, time_info, status):
    volume_norm = np.linalg.norm(indata) / len(indata)
    q_audio.put(volume_norm)

def trigger_action():
    print("🔴 AÇÃO DETECTADA: executando comando...")
    subprocess.Popen(["say", "Comando reconhecido"])  # Exemplo com TTS
    # subprocess.call(["python", "meu_script.py"])
    # subprocess.call(["logline", "span", "trigger", "--evento=voz"])

def run_watchdog():
    print("🎤 Iniciando microfone always-on (Pressione Ctrl+C para sair)")
    last_trigger_time = 0

    with sd.InputStream(callback=audio_callback, channels=1,
                        samplerate=SAMPLE_RATE, blocksize=int(SAMPLE_RATE * CHUNK_DURATION)):
        while True:
            try:
                volume = q_audio.get(timeout=1)
                if volume > THRESHOLD:
                    now = time.time()
                    if now - last_trigger_time > SILENCE_SECONDS:
                        trigger_action()
                        last_trigger_time = now
            except queue.Empty:
                pass

if __name__ == "__main__":
    run_watchdog()
```

import numpy as np
import os

encoder = VoiceEncoder()

# Pré-cadastrados: gravações curtas com 3-5s de fala
voiceprints = {}
for file in os.listdir("voice_profiles"):
    if file.endswith(".wav"):
        name = file.replace(".wav", "")
        wav = preprocess_wav(f"voice_profiles/{file}")
        embed = encoder.embed_utterance(wav)
        voiceprints[name] = embed
```

### 2\. Loop de escuta com VAD + comparação de embedding

```python
import sounddevice as sd
from silero_vad import VoiceActivityDetector
import time

vad = VoiceActivityDetector(sampling_rate=16000)
CHUNK = 16000  # 1 segundo

def recognize_loop():
    print("🎤 Microfone ativo para identificação de voz")
    with sd.InputStream(samplerate=16000, channels=1, dtype='float32') as stream:
        while True:
            audio_chunk = stream.read(CHUNK)[0].flatten()
            if vad.is_speech(audio_chunk):
                wav = np.copy(audio_chunk)
                embed = encoder.embed_utterance(wav)

                # Comparar com todos os perfis
                best_match = None
                best_score = -1
                for name, ref in voiceprints.items():
                    sim = np.dot(embed, ref)  # cosine similarity
                    if sim > best_score:
                        best_score = sim
                        best_match = name

                if best_score > 0.75:
                    print(f"🧠 Reconhecido: {best_match} ({best_score:.2f})")
                    subprocess.call(["logline", "span", "dispatch", f"--speaker={best_match}"])
                else:
                    print(f"🤔 Voz não reconhecida ({best_score:.2f})")

            time.sleep(0.2)
```
