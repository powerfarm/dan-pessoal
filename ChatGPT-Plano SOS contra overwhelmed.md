                        output: input.clone(),
                    });
                }
            }

    use serde::Serialize;
    use serde_json::json;

    #[derive(Debug, Serialize)]
    pub struct NdjsonLine<'a, T: Serialize> {
        pub ts: String,
        pub level: &'a str,
        pub trace_id: &'a str,
        pub event: &'a str,
        pub data: T,
    }

    /// Serializa um evento em NDJSON (string com \\n no final)
    pub fn ndjson<T: Serialize>(level: &str, trace_id: &str, event: &str, data: T) -> String {
        let line = NdjsonLine {
            ts: Utc::now().to_rfc3339_opts(chrono::SecondsFormat::Millis, true),
            level,
            trace_id,
            event,
            data,
        };
        format!("{}\\n", serde_json::to_string(&line).unwrap())
    }

    /// Atalho para log simples com mensagem.
    pub fn ndjson_msg(level: &str, trace_id: &str, msg: &str) -> String {
        ndjson(level, trace_id, "msg", json!({ "message": msg }))
    }
"""))

    use anyhow::Result;
    use policy_engine::{evaluate, Policy, PolicyDecision};
    use serde::{Deserialize, Serialize};
    use serde_json::Value;
    use uuid::Uuid;

    #[derive(Debug, Clone, Serialize, Deserialize)]
    pub struct ExecutionResult {
        pub trace_id: String,
        pub decision: String,
        pub output: Value,
        pub logs: Vec<String>, // NDJSON lines
    }

    pub fn run(input: &Value, policy: &Policy) -> Result<ExecutionResult> {
        let trace_id = Uuid::new_v4().to_string();
        let mut logs = vec![observability::ndjson_msg("INFO", &trace_id, "runtime.start")];
        let PolicyDecision { decision, output } = evaluate(input, policy)?;
        logs.push(observability::ndjson_msg("INFO", &trace_id, &format!("decision:{decision}")));
        logs.push(observability::ndjson_msg("INFO", &trace_id, "runtime.end"));
        Ok(ExecutionResult { trace_id, decision, output, logs })
    }

    #[cfg(test)]
    mod tests {
        use super::*;
        use serde_json::json;

        }

        for r in &policy.rules {
            if let Some(v) = input.get(&r.key) {
                if v == &r.expected {
                    return Ok(PolicyDecision {
                        decision: if r.decision.is_empty() { "allow".into() } else { r.decision.clone() },
                        output: input.clone(),

                }

    from dataclasses import dataclass
    from typing import Dict, List

    import torch
    from torch.utils.data import Dataset

    from transformers import AutoModelForCausalLM, AutoTokenizer, DataCollatorForLanguageModeling, get_linear_schedule_with_warmup

    from peft import LoraConfig, get_peft_model, prepare_model_for_kbit_training

    from src.ndjson_logger import NDJSONLogger

    # ---------- Dataset ----------
    class JsonlInstrDataset(Dataset):
        def __init__(self, path: str, tokenizer, max_length: int, truncate_longer: bool):
            self.samples = []
            with open(path, "r", encoding="utf-8") as f:
                for line in f:
                    if not line.strip():
                        continue
                    obj = json.loads(line)
                    text = f"Instrução: {obj['instruction']}\\nResposta:"
                    target = obj['output']
                    full = text + " " + target
                    tok = tokenizer(full, max_length=max_length, truncation=truncate_longer, padding=False, return_tensors=None)
                    self.samples.append(torch.tensor(tok['input_ids'], dtype=torch.long))
        def __len__(self):
            return len(self.samples)
        def __getitem__(self, idx):
            ids = self.samples[idx]
            return {"input_ids": ids, "labels": ids.clone()}

    def collate_fn(batch, pad_id):
        # Simple left-pad
        maxlen = max(len(x["input_ids"]) for x in batch)
        input_ids = []
        labels = []
        for x in batch:
            ids = x["input_ids"]
            pad_len = maxlen - len(ids)
            if pad_len > 0:
                pad = torch.full((pad_len,), pad_id, dtype=torch.long)
                ids = torch.cat([ids, pad], dim=0)
            input_ids.append(ids)
            labels.append(ids.clone())
        return {"input_ids": torch.stack(input_ids), "labels": torch.stack(labels), "attention_mask": (torch.stack(input_ids) != pad_id).long()}

    # ---------- Train ----------
    def main(cfg_path: str):
        with open(cfg_path, "r", encoding="utf-8") as f:
            cfg = yaml.safe_load(f)

        device = torch.device("mps" if torch.backends.mps.is_available() else "cpu")
        torch.manual_seed(cfg["train"]["seed"])

        logger = NDJSONLogger(cfg["ndjson_path"], cfg.get("trace_id") or "auto")
        logger.log("INFO", "train.start", device=str(device), model_name=cfg["model_name"])

        tokenizer = AutoTokenizer.from_pretrained(cfg["model_name"])
        if tokenizer.pad_token_id is None:
            tokenizer.pad_token = tokenizer.eos_token

        model = AutoModelForCausalLM.from_pretrained(cfg["model_name"])
        model.to(device)
        model.train()

        if cfg["train"].get("gradient_checkpointing", False):
            model.gradient_checkpointing_enable()

        # LoRA
        lora_cfg = cfg.get("lora", {})
        peft_cfg = LoraConfig(
            r=lora_cfg.get("r", 8),
            lora_alpha=lora_cfg.get("alpha", 16),
            lora_dropout=lora_cfg.get("dropout", 0.05),
            target_modules=lora_cfg.get("target_modules", ["q_proj","v_proj"]),
            bias="none",
            task_type="CAUSAL_LM",
        )
        model = get_peft_model(model, peft_cfg)

        # Data
        ds = JsonlInstrDataset(cfg["data_path"], tokenizer, cfg["max_length"], cfg.get("truncate_longer_samples", True))

        # Optimizer & schedule
        lr = cfg["train"]["learning_rate"]
        wd = cfg["train"]["weight_decay"]
        optim = AdamW(model.parameters(), lr=lr, weight_decay=wd)

        # Steps
        per_device_bs = cfg["train"]["batch_size_per_device"]
        gas = cfg["train"]["gradient_accumulation_steps"]
        epochs = cfg["train"]["epochs"]

        # Simple manual loop (for clarity/control on MPS)
        total_steps = math.ceil(len(ds) / per_device_bs / 1) * epochs
        warmup_steps = int(total_steps * cfg["train"]["warmup_ratio"])
        scheduler = get_linear_schedule_with_warmup(optim, num_warmup_steps=warmup_steps, num_training_steps=total_steps)

        scaler = torch.cuda.amp.GradScaler(enabled=False)  # No CUDA here; keep API aligned

        pad_id = tokenizer.pad_token_id
        step = 0
        os.makedirs(cfg["output_dir"], exist_ok=True)
        # save config used
        with open(os.path.join("outputs", "config_used.yaml"), "w", encoding="utf-8") as f:
            yaml.safe_dump(cfg, f, allow_unicode=True)

        for epoch in range(epochs):
            # mini dataloader
            for i in range(0, len(ds), per_device_bs):
                batch = [ds[j] for j in range(i, min(i+per_device_bs, len(ds)))]
                batch = collate_fn(batch, pad_id)
                input_ids = batch["input_ids"].to(device)
                attention_mask = batch["attention_mask"].to(device)
                labels = batch["labels"].to(device)

                with torch.autocast(device_type=("mps" if device.type=="mps" else "cpu"), dtype=torch.float16, enabled=bool(cfg["train"].get("fp16", True) and device.type=="mps")):
                    outputs = model(input_ids=input_ids, attention_mask=attention_mask, labels=labels)
                    loss = outputs.loss / gas

                loss.backward()

                if (step + 1) % gas == 0:
                    optim.step()
                    scheduler.step()
                    model.zero_grad(set_to_none=True)

                if step % cfg["log_every"] == 0:
                    logger.log("INFO", "train.step", step=step, epoch=epoch, loss=float(loss.item()*gas))

                if cfg["save_every_steps"] and step>0 and step % cfg["save_every_steps"] == 0:
                    ckpt_path = os.path.join(cfg["output_dir"], f"lora_step_{step}.pt")
                    model.save_pretrained(ckpt_path)
                    logger.log("INFO", "train.save", path=ckpt_path)

                step += 1

        # final save
        ckpt_path = os.path.join(cfg["output_dir"], "lora_final.pt")
        model.save_pretrained(ckpt_path)
        logger.log("INFO", "train.end", steps=step, final_ckpt=ckpt_path)
